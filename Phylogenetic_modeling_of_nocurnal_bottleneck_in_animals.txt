# =============================================================================
# Pipeline: Multispecies Gene Phylogeny Analysis for Nocturnal vs Diurnal Traits
# Author: Susanta Behura
#
# Description:
# This pipeline models gene phylogenies across multiple mammalian species and
# evaluates whether gene-specific evolutionary histories are associated with
# species-level activity patterns (nocturnal vs diurnal).
#
# Workflow:
#   1) Download coding sequences (CDS) for multiple species
#   2) Extract the longest CDS per gene for each species
#   3) Retain annotated genes shared across species
#   4) Generate multiple sequence alignments
#   5) Build gene phylogenies
#   6) Root trees using an outgroup
#   7) Convert trees to Nexus format
#   8) Perform trait-phylogeny modeling with BayesTraits
#   9) Compare dependent vs independent models using log Bayes factors
#
# Reference:
# Poudel SP, Behura SK. Relevance of the regulation of the brain-placental axis
# to the nocturnal bottleneck of mammals. Placenta. 2024;155:11-21.
#
# Requirements:
# - wget, gunzip
# - awk, sed, sort
# - Clustal Omega (clustalo)
# - phyx (pxnj, pxrr, pxt2nex)
# - BayesTraitsV4
#
# Input:
# - Ensembl CDS FASTA files from multiple mammalian species
# - Trait file encoding nocturnal/diurnal states
# - BayesTraits command files for dependent and independent models
#
# Output:
# - Longest CDS per gene per species
# - Gene-specific sequence alignments
# - Gene phylogenies in Newick and Nexus formats
# - Log marginal likelihoods for dependent and independent models
# - Candidate genes whose phylogenies are associated with activity pattern
# =============================================================================

# Decompress all downloaded FASTA files
gunzip *.fa.gz


# =============================================================================
# STEP 2: Process CDS FASTA Files
# =============================================================================

# Retain gene symbol in FASTA headers
for i in *.cds.all.fa; do
    sed 's/^.*gene_symbol:/>/' "$i" | awk '{print $1}' > "${i}_name"
done

# Convert multiline FASTA to single-line FASTA
for i in *_name; do
    awk '{
        if (NR==1) print $0;
        else if ($0 ~ /^>/) print "\n"$0;
        else printf $0
    }' "$i" > "${i}_single"
done

# Convert FASTA to tabular format
for i in *_single; do
    awk -v OFS="\t" 'BEGIN{RS=">"} {
        gsub("\n","\t",$0);
        gsub(">","",$0);
        print $0
    }' "$i" > "${i}_table"
done

# Retain the longest CDS per gene
for i in *_table; do
    awk '{print $1, $2, length($2)}' "$i" | \
        sort -k1,1 -k3,3nr | \
        sort -u -k1,1 | \
        awk -v OFS='\t' '{print $1, $2}' > "${i}_longest"
done


# =============================================================================
# STEP 3: Combine Gene Sequences Across Species
# =============================================================================

awk '{print $0, FILENAME}' *.cds.all.fa_name_single_table_longest > all_genes_sequences_species

# Remove unannotated genes (retain known gene symbols only)
awk '!/ENS/' all_genes_sequences_species > all_known_genes_sequences_species

# Retain unique gene-species combinations
awk -F"\t" '!seen[$1, $3]++' all_known_genes_sequences_species > uniq_gene_seq_species


# =============================================================================
# STEP 4: Generate Per-Gene Sequence Files Across Species
# =============================================================================

# Create one FASTA file per gene for downstream alignment
for gene in $(awk '{print $1}' uniq_gene_seq_species | sort | uniq); do
    grep -w "$gene" uniq_gene_seq_species | \
        awk -v OFS='\n' '{print ">"$3, $2}' > "seq_${gene}.fa"
done


# =============================================================================
# STEP 5: Perform Multiple Sequence Alignment
# =============================================================================

module load clustalo

for i in seq_*.fa; do
    clustalo -i "$i" -o "aln_${i}" --force
done


# =============================================================================
# STEP 6: Generate Gene Phylogenies
# =============================================================================

module load phyx

for i in aln_seq_*.fa; do
    pxnj -s "$i" -o "newick_${i}"
done


# =============================================================================
# STEP 7: Root Phylogenetic Trees
# =============================================================================

# Root each tree using opossum as the outgroup
for i in newick_aln_seq_*.fa; do
    pxrr -t "$i" -g "Opossum" -r -o "root_${i}"
done


# =============================================================================
# STEP 8: Convert Newick Trees to Nexus Format
# =============================================================================

for i in root_newick_aln_seq_*.fa; do
    pxt2nex -t "$i" -o "nexus_${i}"
done


# =============================================================================
# STEP 9: Perform Trait-Phylogeny Modeling with BayesTraits
# =============================================================================

module load BayesTraitsV4.0.0

# Dependent model: gene phylogeny and trait evolution are linked
for i in nexus_*; do
    BayesTraitsV4 "$i" traitfile < discrete_dependent_model
    grep -w "Log marginal likelihood" traitfile.Stones.txt > "LML_DEP_${i}.txt"
done

awk '{print $0, FILENAME}' LML_DEP_* | \
    awk -v OFS='\t' '{print $5, $4, $NF}' | \
    sed 's/.txt//g' | \
    grep -v "inf" > All_trees_logMarginalLikelihoods_discrete_DEPENDENT_model


# Independent model: gene phylogeny and trait evolution are independent
for i in nexus_*; do
    BayesTraitsV4 "$i" traitfile < discrete_independent_model
    grep -w "Log marginal likelihood" traitfile.Stones.txt > "LML_INDEP_${i}.txt"
done

awk '{print $0, FILENAME}' LML_INDEP_* | \
    awk -v OFS='\t' '{print $5, $4, $NF}' | \
    sed 's/.txt//g' | \
    grep -v "inf" > All_trees_logMarginalLikelihoods_discrete_INDEPENDENT_model


# =============================================================================
# STEP 10: Compare Models Using Log Bayes Factors
# =============================================================================

awk -F'\t' -v OFS='\t' '
NR==FNR {dep[$3]=$1; next}
($3 in dep) {
    lbf = 2 * (dep[$3] - $1);
    print $3, dep[$3], $1, lbf
}' All_trees_logMarginalLikelihoods_discrete_DEPENDENT_model \
   All_trees_logMarginalLikelihoods_discrete_INDEPENDENT_model \
   > log_bayes_factor_gene_trait_association.txt
